\section{编译和链接过程}
\begin{enumerate}
    \item 源文件：.c
    \item 编译之后 $=>$ 目标文件：.o（object file）
    \item 目标文件打包 $=>$ 归档文件：.a（archive file）
    \item 链接之后 $=>$ 可执行文件（没有后缀）
\end{enumerate}

\section{一个简单的Makefile文件}

    $(MAKE) -C $(KDIR) SUBDIRS=$(PWD) modules
.PHONY:clean
clean:
    rm -rf *.o *.order *.mod.c *.symvers
\end{lstlisting}
